49      @Override
50      protected JExpression collectionStep1(UserType arg, JClass collectionClass) {
51          String utname = arg.getTypeName();
52          String utnamec = camelize(utname);
53          JClass sourceCollectionGeneric = jcm.ref(getFullClassName(jpackage, utnamec));
54          JClass sourceCollectionClass = jcm.ref(arg.asJavaClass())
55                  .narrow(sourceCollectionGeneric);
56  
57          JVar source = body.decl(sourceCollectionClass, fnamecl + "Source",
58                  data.invoke("get" + camelize(fname)));
59          JVar target = body.decl(jcm.ref(arg.asJavaClass()).narrow(UDTValue.class),
60                  fnamecl + "Target",
61                  JExpr._new(arg.asJavaClass().isAssignableFrom(Set.class)
62                          ? jcm.ref(HashSet.class).narrow(UDTValue.class)
63                          : jcm.ref(LinkedList.class).narrow(UDTValue.class)));
64          JForEach forEach = body.forEach(sourceCollectionGeneric, "entry", source);
65          JVar entry = forEach.var();
66          JBlock forEachBody = forEach.body();
67  
68          JInvocation convertToUDT = sourceCollectionGeneric
69                  .staticInvoke("udtMapper").invoke("toUDT").arg(entry).arg(session);
70          forEachBody.add(target.invoke("add").arg(convertToUDT));
71          return target;
72      }

79      @Override
80      protected JExpression mapStep1(DataType arg0, DataType arg1, JClass argc0, JClass argc1) {
81          JVar source = body.decl(jcm.ref(Map.class).narrow(argc0, argc1),
82                  fnamecl + "Source", data.invoke("get" + camelize(fname)));
83          JVar target = body.decl(jcm.ref(Map.class).narrow(arg0.asJavaClass(), arg1.asJavaClass()),
84                  fnamecl + "Target",
85                  JExpr._new(jcm.ref(HashMap.class).narrow(arg0.asJavaClass(), arg1.asJavaClass())));
86          JClass entryClass = jcm.ref(Map.Entry.class).narrow(argc0, argc1);
87          JForEach forEach = body.forEach(entryClass, "entry", source.invoke("entrySet"));
88          JVar entry = forEach.var();
89          JExpression k = entry.invoke("getKey");
90          JExpression v = entry.invoke("getValue");
91          JBlock forEachBody = forEach.body();
92          JExpression key = arg0.isFrozen()
93                  ? argc0.staticInvoke("udtMapper").invoke("toUDT").arg(k).arg(session)
94                  : k;
95          JExpression value = arg1.isFrozen()
96                  ? argc1.staticInvoke("udtMapper").invoke("toUDT").arg(v).arg(session)
97                  : v;
98          forEachBody.add(target.invoke("put").arg(key).arg(value));
99          return target;
100     }
